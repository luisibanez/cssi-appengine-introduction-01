main.py


import webapp2
            

class MainHandler(webapp2.RequestHandler):
    def get(self):
        self.response.write('Hello world!')
            

app = webapp2.WSGIApplication([
    ('/', MainHandler)
], debug=True)

main.py


class CountHandler(webapp2.RequestHandler):
    def get(self):
        for i in range(1, 21):
          self.response.write('Hello %d 
' % i)
            

app = webapp2.WSGIApplication([
    ('/', MainHandler),
    ('/count', CountHandler)
], debug=True)

You Do

Modify the Code

  • In main.py
    • Add CountHandler
    • Add /count route
  • Refresh the page http://localhost:8080
  • Open the page http://localhost:8080/count
